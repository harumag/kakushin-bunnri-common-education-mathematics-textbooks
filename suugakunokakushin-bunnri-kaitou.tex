\begin{center}

  \end{tabular}

\reibanni $\wa{i=1}{n}\dfrac{1}{i^2}<2-\dfrac{1}{n}$を示す.\\
\kangaekata\\
一般の$n$に対する証明なので,帰納法で攻めるのがよいでしょう.\\
\hokukai 数学的帰納法で$\wa{i=1}{n}\dfrac{1}{i^2}<2-\dfrac{1}{n}\cdots\cdots\asta$を示す.\\
今,$n\geq2$なので,\\
\tokeiichi\underline{$n=2$のとき,}\\
$\asta$ は,\begin{align*}
  1+\dfrac{1}{4}<2-\dfrac{1}{2}\doti \dfrac{5}{4}<\dfrac{3}{2}
\end{align*}
となって$\asta$は成立する.\\
\tokeini\underline{$n=k　(k\geq2)$のとき,}\\
$\wa{i=1}{k}\dfrac{1}{i^2}<2-\dfrac{1}{k}\cdots\cdots\maruichi$が成立すると仮定して,
$\wa{k=1}{k+1}\dfrac{1}{i^2}<2-\dfrac{1}{k+1}\cdots\cdots\maruni$を示す.
\begin{align*}
  \maruni\doti\wa{i=1}{k}\dfrac{1}{i^2}+\dfrac{1}{\p{k+1}^2}<2-\dfrac{1}{k}-\p{\dfrac{1}{k+1}-\dfrac{1}{k}}
\end{align*}
と変形でき,$k\in\mathbb{Z}\geq2$であり,
\begin{align*}
  \dfrac{1}{(k+1)^2}<\dfrac{1}{k(k+1)}
\end{align*}
なので,$\maruichi$の両辺に辺々足して,
\begin{align*}
  \wa{i=1}{k}\dfrac{1}{i^2}+\dfrac{1}{(k+1)^2}&<2-\dfrac{1}{k}+\p{\dfrac{1}{k}-\dfrac{1}{k+1}}\\
  &=2-\dfrac{1}{k+1}
\end{align*}
$\Y \wa{i=1}{k+1}\dfrac{1}{i^2}<2-\dfrac{1}{k+1}.$よって,$n=k+1$のときも$\asta$は成立する.\\
以上より,$すべての2以上の自然数に対して,\asta$は成立する.　$\blacksquare$

\noindent\reibansan$5^n<2^n\cdot n(n+2)$を示す.\\
\kangaekata
こういう問題はまず実験しましょう.実験してみると,次のような規則がわかります.
\begin{align*}
  &n=1のとき,5<2\cdot1\cdot3=6\\
  &n=2のとき,25<4\cdot2\cdot4=32\\
  &n=3のとき,125<8\cdot3\cdot5=120\\
  &\vdots
\end{align*}
よって,$\ans{n=1,2}$が答えとわかります.あとは,それを正当化しにくための証明を考えれば良いのです.
\hokukai $5^n<2^n\cdot n(n+2)\doti 1<\p{\dfrac{2}{5}}^n\cdot n(n+2)\cdots\cdots\asta $を示す.\\
$\asta $の右辺を$f(n)　(n\geq1)$とおいて,$f(n>1)$を満たす$n$を求める.
\begin{align*}
  \Delta f(n)&=\p{\dfrac{2}{5}}^{n+1}\cdot\p{n+1}\p{n+3}-\p{\dfrac{2}{5}}^n\cdot n(n+2)\\
  &=\p{\dfrac{2}{5}}^{n}\B{\dfrac{2}{5}(n+1)(n+3)-n(n+2)}\\
  &=-\dfrac{1}{5}\p{\dfrac{2}{5}}^n(3n^2+2n-6)
\end{align*}
ここで,$g(n)=3n^2+2n-6　(n\geq1)$とおくと,$g(n)=n(3n+2)-6$より,$g(n)$は$n\geq1$で単調増加で,$g(1)=-1,g(2)=10$より,
\begin{align*}
  g(1)<0<g(2)
\end{align*}
なので,
\begin{align*}
  g(1)<0<g(2)<g(3)<g(4)<\cdots
\end{align*}
であることがわかる.よって,
\begin{align*}
  \begin{cases}
    n=1のとき,\Delta f(n)>0\\
    n\geq2のとき,\Delta f(n)<0
  \end{cases}
\end{align*}
より,
\begin{align*}
  f(1)<f(2)>f(3)>f(4)>\cdots
\end{align*}
ここで,
\begin{align*}
  f(1)=\dfrac{6}{5}>1,　f(3)=\dfrac{24}{25}<1
\end{align*}
であるから,$f(3)$以降は1より小さい.\\
$\Y 求める,nは,\ans{n=1,2}$
